<Steps>

<Step>Run the following command:</Step>

```bash
npx shadcn@latest add https://ui-x.junwen-k.dev/r/bprogress-provider-next-app.json
```

<Step>
  Wrap your root layout with the `<BProgressProvider />` component:
</Step>

```tsx title="app/layout.tsx"
/* [!code highlight] */
import { BProgressProvider } from "@/components/bprogress-provider";

export default function RootLayout({ children }: RootLayoutProps) {
  return (
    <html lang="en">
      <head />
      <body>
        {/* [!code highlight] */}
        <BProgressProvider>{children}</BProgressProvider>
      </body>
    </html>
  );
}
```

</Steps>
