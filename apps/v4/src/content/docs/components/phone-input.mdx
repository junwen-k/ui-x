## Usage

```tsx
import { PhoneInput } from "@/components/ui/phone-input";
```

```tsx
<PhoneInput>
  <PhoneInputCountrySelect>
    <PhoneInputCountrySelectTrigger>
      <PhoneInputCountrySelectValue />
    </PhoneInputCountrySelectTrigger>
    <PhoneInputCountrySelectContent>
      <PhoneInputCountrySelectOptions />
    </PhoneInputCountrySelectContent>
  </PhoneInputCountrySelect>
  <PhoneInputInput />
</PhoneInput>
```
