## Usage

```tsx
import {
  EmojiPicker,
  EmojiPickerSearch,
  EmojiPickerContent,
  EmojiPickerFooter,
} from "@/components/ui/emoji-picker";
```

```tsx
<EmojiPicker>
  <EmojiPickerSearch />
  <EmojiPickerContent />
  <EmojiPickerFooter />
</EmojiPicker>
```
